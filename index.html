Programación

El siguiente es el código utilizado para el robot seguidor de línea:

        #include 

            // Pines de los motores (usando pines analógicos para el puente H)
            const int motorIzqA = A0;   // Entrada del puente H para el motor izquierdo
            const int motorIzqB = A1;
            const int motorDerA = A2;   // Entrada del puente H para el motor derecho
            const int motorDerB = A3;
            
            // Configuración de los sensores
            QTRSensors qtr;
            const uint8_t SensorCount = 8;
            uint16_t sensorValues[SensorCount];
            
            // Constantes para el control PID
            int velocidadBase = 150;      // Velocidad base de los motores
            float kp = 0.1;               // Ajusta la constante proporcional
            float kd = 0.3;               // Ajusta la constante derivativa
            int ultimoError = 0;
            
            void setup() {
              // Configura los sensores
              qtr.setTypeRC();
              qtr.setSensorPins((const uint8_t[]){3, 4, 5, 6, 7, 8, 9, 10}, SensorCount);
              qtr.setEmitterPin(2);
            
              // Configura los pines de los motores como salida
              pinMode(motorIzqA, OUTPUT);
              pinMode(motorIzqB, OUTPUT);
              pinMode(motorDerA, OUTPUT);
              pinMode(motorDerB, OUTPUT);
            
              Serial.begin(9600);
              delay(1000);
              Serial.println("Calibrando sensores...");
            
              // Calibración de los sensores
              for (int i = 0; i < 400; i++) {
                qtr.calibrate();
                delay(20);
              }
              Serial.println("Calibración completada");
            }
            
            void loop() {
              // Lee la posición de la línea
              int posicion = qtr.readLineBlack(sensorValues);
              Serial.print("Posición: ");
              Serial.println(posicion);  // Imprime la posición para depuración
              
              int error = posicion - 3500;  // Ajuste el valor de 3500 según los datos de depuración
            
              // Cálculo del PID
              int ajuste = kp * error + kd * (error - ultimoError);
              ultimoError = error;
            
              // Ajusta el valor de "ajuste" para limitar la respuesta del robot
              if (ajuste > 100) {
                ajuste = 100;
              } else if (ajuste < -100) {
                ajuste = -100;
              }
            
              // Calcula la velocidad de cada motor
              int velocidadIzquierda = velocidadBase + ajuste;
              int velocidadDerecha = velocidadBase - ajuste;
            
              // Limita la velocidad entre 0 y 255
              velocidadIzquierda = constrain(velocidadIzquierda, 0, 255);
              velocidadDerecha = constrain(velocidadDerecha, 0, 255);
            
              // Mensaje de depuración para indicar la dirección
              if (ajuste > 0) {
                Serial.println("Girando a la derecha");
              } else if (ajuste < 0) {
                Serial.println("Girando a la izquierda");
              } else {
                Serial.println("Moviéndose hacia adelante");
              }
            
              // Controla los motores
              moverMotorIzquierdo(velocidadIzquierda);
              moverMotorDerecho(velocidadDerecha);
              
              delay(100);  // Pequeño retraso para evitar lecturas demasiado rápidas
            }
            
            // Funciones para controlar los motores
            void moverMotorIzquierdo(int velocidad) {
              if (velocidad > 0) {
                analogWrite(motorIzqA, velocidad);
                analogWrite(motorIzqB, 0);
              } else {
                analogWrite(motorIzqA, 0);
                analogWrite(motorIzqB, -velocidad);
              }
            }
            
            void moverMotorDerecho(int velocidad) {
              if (velocidad > 0) {
                analogWrite(motorDerA, velocidad);
                analogWrite(motorDerB, 0);
              } else {
                analogWrite(motorDerA, 0);
                analogWrite(motorDerB, -velocidad);
              }
            }
